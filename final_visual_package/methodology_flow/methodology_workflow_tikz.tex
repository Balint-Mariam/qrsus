\newcommand{\workflowbox}[5]{%
  \node[box=#4] (#1) at (#2) {%

    {\bfseries #3}\\[2pt]
    #5
  };
  \node[titlebox=#4, anchor=north] at ([yshift=-1.5pt]#1.north) {#3};
}
