\tikzset{
  box/.style={
    rounded corners=6pt,
    draw=#1,
    very thick,
    fill=white,
    text=DarkGray,
    align=center,
    text width=4.45cm,
    inner xsep=8pt,
    inner ysep=10pt,

    minimum height=1.70cm,
    font=\footnotesize
  },
  titlebox/.style={
    font=\bfseries\footnotesize,

    inner xsep=6pt,
    inner ysep=3pt
  },
  layerlabel/.style={
    font=\bfseries\scriptsize,
    text=#1,
    align=left
  },
  flowarrow/.style={
    -{Stealth[length=3.0mm,width=2.0mm]},

  },
  softpanel/.style={
    rounded corners=8pt,

    draw=MidGray,
    line width=0.4pt
  }
}

\newcommand{\workflowbox}[5]{%
  \node[box=#4] (#1) at (#2) {%
    #5
  };
  \node[titlebox=#4, anchor=south] at ([yshift=2.5pt]#1.north) {#3};
}
